let one = document.querySelector("#first"); - Select the first tag
one.style = "font:40pt Arial;";
let para = document.querySelectorAll(".para");
para[1].style = "color:red";
let all = document.querySelectorAll(".para");
for ( let x of all){
x.style.setProperty("background-color","yellow"); ->Sets background for all para tags
}
